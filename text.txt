    function startCountdownTimer() {
      if (activeCountdownInterval) clearInterval(activeCountdownInterval);
      activeCountdownInterval = setInterval(() => {
        const timerBoxes = document.querySelectorAll('.match-countdown-box');
        timerBoxes.forEach(el => {
          const matchTime = parseInt(el.getAttribute('data-match-time'));
          if (isNaN(matchTime) || matchTime <= 0) {
            el.innerHTML = `<span style="color: var(--text-muted);"><i class="fa-regular fa-clock"></i> Time Undefined</span>`;
            return;
          }

          const diff = matchTime - Date.now();
          if (diff > 0) {
            const days = Math.floor(diff / (1000 * 60 * 60 * 24));
            const hrs = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const mins = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
            const secs = Math.floor((diff % (1000 * 60)) / 1000);
            
            let timeStr = "";
            if (days > 0) timeStr += `${days} দিন `;
            if (hrs > 0 || days > 0) timeStr += `${hrs} ঘণ্টা `;
            timeStr += `${mins} মিনিট ${secs} সেকেন্ড`;
            
            el.innerHTML = `<span style="color: #ff9f43; font-weight: 600;"><i class="fa-regular fa-clock"></i> Starts in: <strong style="color: var(--neon-blue);">${timeStr}</strong></span>`;
          } else {
            const postTime = Math.abs(diff);
            if (postTime < 3600000) { // Keep Live status for 1 hour
              el.innerHTML = `<span style="color: var(--neon-red); font-weight:bold; animation: pulse 1s infinite;"><i class="fa-solid fa-circle-dot"></i> MATCH IS LIVE</span>`;
            } else {
              el.innerHTML = `<span style="color: var(--text-muted);"><i class="fa-solid fa-circle-check"></i> Match Started</span>`;
            }
          }
        });
      }, 1000);
    }

    // 🎮 SUBMIT JOIN MATCH FUNCTION
    function submitJoinMatch(e) {
      e.preventDefault();
      if (!currentUser) return;
      const matchId = document.getElementById('joining-match-id').value;
      const fee = parseInt(document.getElementById('joining-match-fee').value) || 0;
      const match = tData[matchId];
      if (!match) return;

      const submitBtn = e.target.querySelector('button[type="submit"]');
      if (submitBtn) {
        submitBtn.disabled = true;
        submitBtn.style.opacity = "0.5";
        submitBtn.innerText = "PROCESSING...";
      }

      const mt = match.matchType || "solo";
      const teamCount = mt === "squad" ? 4 : (mt === "duo" ? 2 : 1);
      const players = [];

      for (let i = 1; i <= teamCount; i++) {
        const nick = document.getElementById(`join-n-${i}`).value.trim();
        const uid = document.getElementById(`join-u-${i}`).value.trim();
        if (!nick || !uid) {
          showErrorModal('Error!', 'দয়া করে সব খেলোয়াড়ের নাম এবং UID সঠিক তথ্য দিন।');
          if (submitBtn) { submitBtn.disabled = false; submitBtn.style.opacity = "1"; submitBtn.innerText = "CONFIRM REGISTRATION"; }
          return;
        }
        players.push({ name: nick, uid: uid });
      }

      // Check current wallet balance dynamically
      db.ref('users/' + currentUser.uid).once('value', uSnap => {
        const user = uSnap.val();
        if (!user || (user.balance || 0) < fee) {
          showErrorModal('অপর্যাপ্ত ব্যালেন্স!', `ম্যাচ এন্ট্রি ফি ৳ ${fee} টাকা। অনুগ্রহ করে ওয়ালেট থেকে ব্যালেন্স এড করুন।`);
          if (submitBtn) { submitBtn.disabled = false; submitBtn.style.opacity = "1"; submitBtn.innerText = "CONFIRM REGISTRATION"; }
          return;
        }

        // Check if Slot is Full
        db.ref(`participants/${matchId}`).once('value', pSnap => {
          const currentCount = pSnap.val() ? Object.keys(pSnap.val()).length : 0;
          if (currentCount >= match.maxPlayers) {
            showErrorModal('Slot Full!', 'দুঃখিত, এই টুর্নামেন্টের সকল স্লট পূর্ণ হয়ে গিয়েছে!');
            if (submitBtn) { submitBtn.disabled = false; submitBtn.style.opacity = "1"; submitBtn.innerText = "CONFIRM REGISTRATION"; }
            return;
          }

          // Abort if already joined
          let alreadyJoined = false;
          if (pSnap.exists()) {
            const parts = pSnap.val();
            if (parts[currentUser.uid]) alreadyJoined = true;
            else {
              for (let key in parts) {
                if (parts[key].players && parts[key].players.some(p => p.uid === currentUser.uid)) {
                  alreadyJoined = true; break;
                }
              }
            }
          }

          if (alreadyJoined) {
            showErrorModal('ইতিমধ্যে নিবন্ধিত!', 'আপনি ইতিমধ্যে এই ম্যাচে সফলভাবে অংশগ্রহণ করেছেন।');
            closePopup('join-popup-overlay');
            if (submitBtn) { submitBtn.disabled = false; submitBtn.style.opacity = "1"; submitBtn.innerText = "CONFIRM REGISTRATION"; }
            return;
          }

          // Atomic Balance Transaction Deduct Block
          db.ref('users/' + currentUser.uid).transaction(userData => {
            if (userData) {
              if ((userData.balance || 0) < fee) return; 
              userData.balance = (userData.balance || 0) - fee;
            }
            return userData;
          }, (txError, committed) => {
            if (txError || !committed) {
              showErrorModal('ত্রুটি!', 'পেমেন্ট ট্রানজেকশনে কোনো সমস্যা হয়েছে। আবার চেষ্টা করুন।');
              if (submitBtn) { submitBtn.disabled = false; submitBtn.style.opacity = "1"; submitBtn.innerText = "CONFIRM REGISTRATION"; }
            } else {
              // Successfully deducted, write registration data
              const registration = {
                teamLeader: currentUser.uid,
                leaderName: currentUser.name,
                timestamp: Date.now(),
                players: players,
                feePaid: fee,
                slot: currentCount + 1
              };

              db.ref(`participants/${matchId}/${currentUser.uid}`).set(registration).then(() => {
                db.ref(`transactions/${currentUser.uid}`).push().set({
                  type: 'join',
                  label: `${match.code || ''} Match Registration Fee`,
                  amount: fee,
                  timestamp: Date.now()
                });
                closePopup('join-popup-overlay');
                showSuccessModal('নিবন্ধন সফল!', `অভিনন্দন! আপনি সফলভাবে ৳ ${fee} টাকা এন্ট্রি ফির বিনিময়ে "${match.name}" ম্যাচে জয়েন করেছেন।`);
              });
            }
          });
        });
      });
    }

      btn.style.opacity = "0.5";

      db.ref('app_settings/daily_bonus').once('value').then(bSnap => {
        const bonusVal = parseInt(bSnap.val()) || 0;
        if(bonusVal <= 0){ 
          showErrorModal('অক্ষম করা!', 'ডেইলি বোনাস অফারটি বর্তমানে বন্ধ আছে।'); 
          btn.disabled = false; btn.style.opacity = "1"; return; 
        }

        const todayString = new Date().toDateString();

        db.ref('users/' + currentUser.uid).transaction(user => {
          if (user) {
            let lastClaimString = "";
            if (user.last_daily_claim) {
              lastClaimString = new Date(Number(user.last_daily_claim)).toDateString();
            }
            if (todayString === lastClaimString) return; 

            user.balance = (user.balance || 0) + bonusVal;
            user.last_daily_claim = Date.now();
          }
          return user;
        }, (error, committed) => {
          if (error) { showErrorModal('ত্রুটি!', 'বোনাস ক্লেইম করতে ব্যর্থ হয়েছে।'); } 
          else if (!committed) { showErrorModal('আজকে ইতিমধ্যে ক্লেইমড!', 'আজকের বোনাস আপনি নিয়ে নিয়েছেন। আগামীকাল আবার চেষ্টা করুন।'); } 
          else {
            db.ref(`transactions/${currentUser.uid}`).push().set({ 
              type: 'add', label: "Daily Login Bonus", amount: bonusVal, timestamp: Date.now() 
            });
            showSuccessModal('সফল ক্লেইম!', `আপনি ওয়ালেটে ৳ ${bonusVal} টাকা ডেইলি বোনাস পেয়েছেন।`);
          }
          btn.disabled = false; btn.style.opacity = "1";
        });
      });
    }

    function checkUserParticipation(matchId, matchObj) {
      if (!currentUser) return false; const uid = currentUser.uid;
      if (matchId && pData[matchId]) { if (pData[matchId][uid]) return true; for (let regId in pData[matchId]) { const reg = pData[matchId][regId]; if (reg.players && Array.isArray(reg.players)) { if (reg.players.some(p => p.uid === uid)) return true; } } }
      if (matchObj) { if (matchObj.participants) { if (matchObj.participants[uid]) return true; for (let regId in matchObj.participants) { const reg = matchObj.participants[regId]; if (reg.players && Array.isArray(reg.players)) { if (reg.players.some(p => p.uid === uid)) return true; } } } if (matchObj.players && Array.isArray(matchObj.players)) { if (matchObj.players.some(p => p.uid === uid)) return true; } if (matchObj.winner && matchObj.winner.uid === uid) { return true; } }
      return false;
    }

    function loadCompletedMatches() {
      db.ref('match_history').on('value', s => {
        const container = document.getElementById('completed-matches-history-container'); container.innerHTML = '';
        if (s.exists()) {
          let matches = []; s.forEach(child => { let m = child.val(); m.key = child.key; matches.push(m); }); matches.reverse();
          let userMatchesCount = 0;
          matches.forEach(m => {
            const matchId = m.key || m.id; if (!checkUserParticipation(matchId, m)) { return; } userMatchesCount++;
            const winnerBlockHtml = m.winner ? `<div style="background: rgba(16, 185, 129, 0.08); border: 1px dashed #10b981; border-radius: 8px; padding: 10px; margin-top: 10px; font-size:0.82rem;"><strong style="color: #10b981;"><i class="fa-solid fa-trophy"></i> Winner:</strong> ${m.winner.name} (UID: ${m.winner.uid})<br><span style="color:var(--text-muted);">Kills: ${m.winner.kills} | Prize: ${m.winner.prize}</span></div>` : `<p style="font-size: 0.8rem; color: var(--neon-red); margin-top: 10px;"><i class="fa-solid fa-circle-info"></i> Winner details will be declared soon.</p>`;

        } else { container.innerHTML = '<p style="color: var(--text-muted); text-align: center;">No completed matches found.</p>'; }
      });
    }

    function openBannerDetails(id, name) {
      const match = tData[id];
      if (!match) return;
      document.getElementById('banner-popup-match-title').textContent = name;
      document.getElementById('popup-tab-rules').innerHTML = match.rules || "No rules defined yet.";
      
      const jp = pData[id] || {};
      const ul = document.getElementById('popup-joined-players-ul');
      ul.innerHTML = '';
      let count = 1;

      for (let userKey in jp) {
        const regObj = jp[userKey];
        if (regObj.players && Array.isArray(regObj.players)) {
          regObj.players.forEach(p => {
            ul.innerHTML += `<li class="joined-p-item"><span>${count++}. <strong>${p.name}</strong></span><span style="color:var(--text-muted); font-size:0.75rem;">UID: ${p.uid}</span></li>`;
          });
        }
      }
      if (ul.innerHTML === '') {
        ul.innerHTML = '<li class="joined-p-item" style="color: var(--text-muted);">No players registered yet.</li>';
      }
      document.getElementById('banner-details-popup').style.display = 'flex';
      togglePopupTab('players');
    }
